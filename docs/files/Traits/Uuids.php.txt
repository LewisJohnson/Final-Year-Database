namespace SussexProjects\Traits;

use Illuminate\Support\Str;

trait Uuids{
	/**
	 * Boot function from laravel.
	 */
	protected static function boot(){
		parent::boot();

		static::creating(function ($model) {
			$model->{$model->getKeyName()} = (string) Str::orderedUuid();
		});
	}
}
